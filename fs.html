INIT

var fs = ign.filesystem();

SYSTEM PATH

Check Home Path

FUNC : fs.homePath()

Check Application Path

FUNC : fs.appPath()

CHECKING FILE or DIRECTORY

Checking File or Dir exist

FUNC : fs.axists("/PATH/to/file")

OUT : bool

Checking Directory

FUNC : fs.isDirectory("/PATH/to/dir")

OUT : bool

Checking File

FUNC : fs.isFile("/PATH/to/file")

OUT : bool

Checking Directory

FUNC : fs.isDirectory("/PATH/to/dir")

OUT : bool

Checking Absolute Directory

FUNC : fs.isAbsolute("/PATH/to/dir")

OUT : bool

Checking Executable File

FUNC : fs.isExecutable("/PATH/to/file")

OUT : bool

Checking Writable File

FUNC : fs.isWritable("/PATH/to/file")

OUT : bool

Checking Sym Link

FUNC : fs.isLink("/PATH/to/file")

OUT : bool

DIRECTORY HANDLING

FUNC : fs.dir("option","/PATH/to/dir")

OPTIONS :
"remove" : remove directory
"create" : create directory

OUT : bool

FILE HANDLING

File Remove

FUNC : fs.fileRemove("/PATH/to/file")

OUT : bool

File Write

FUNC : fs.fileWrite("/PATH/to/file","input")

INPUT : string

OUT : bool

File Read

FUNC : fs.fileRead("/PATH/to/file")

OUT : string
